// BFS Algorithm
function BFS(graph, start):
    queue = [start]
    visited = set()
    
    while queue is not empty:
        current = queue.dequeue()
        if current not in visited:
            visited.add(current)
            for neighbor in graph[current]:
                if neighbor not in visited:
                    queue.enqueue(neighbor)
